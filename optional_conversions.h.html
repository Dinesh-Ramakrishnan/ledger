   10   namespace detail {
   11 
   12     template <typename T>
   13     struct to_python
   14     {
   15       static PyObject*
   16       convert(
   17         boost::optional<T> const& value)
   18       {
   19         if (value) {
   20           return boost::python::incref(boost::python::object(*value).ptr());
   21         }
   22         return boost::python::incref(Py_None);
   23       }
   24     };
   25 
   26   } // namespace detail
   27 
   28   template <typename T>
   29   struct to_python
   30   {
   31     to_python()
   32     {
   33       boost::python::to_python_converter<
   34         boost::optional<T>,
   35         detail::to_python<T> >();
   36     }
   37   };
   38 
   39   template <typename T>
   40   struct from_python
   41   {
   42     from_python()
   43     {
   44       boost::python::converter::registry::push_back(
   45         &convertible,
   46         &construct,
   47         boost::python::type_id<boost::optional<T> >());
   48     }
   49 
   50     static void* convertible(PyObject* obj_ptr)
   51     {
   52       if (obj_ptr != Py_None) {
   53         boost::python::extract<T> proxy(obj_ptr);
   54         if (!proxy.check()) return 0;
   55       }
   56       return obj_ptr;
   57     }
   58 
   59     static void construct(
   60       PyObject* obj_ptr,
   61       boost::python::converter::rvalue_from_python_stage1_data* data)
   62     {
   63       boost::optional<T> value;
   64       if (obj_ptr != Py_None) {
   65         boost::python::extract<T> proxy(obj_ptr);
   66         value.reset(proxy());
   67       }
   68       void* storage = (
   69         (boost::python::converter::rvalue_from_python_storage<
   70           boost::optional<T> >*)
   71             data)->storage.bytes;
   72       new (storage) boost::optional<T>(value);
   73       data->convertible = storage;
   74     }
   75   };
   76 
   77   template <typename T>
   78   struct to_and_from_python
   79   {
   80     to_and_from_python()
   81     {
   82       to_python<T>();
   83       from_python<T>();
   84     }
   85   };
   86 
   87 }} // namespace boost_adptbx::optional_conversions
